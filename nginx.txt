server {
  listen 80;

  server_name api.ezeugwagerrard.com www.api.ezeugwagerrard.com;              

  access_log /var/log/nginx/api.ezeugwagerrard.com.access.log;
  error_log /var/log/nginx/api.ezeugwagerrard.com.error.log;

  # Redirect HTTP to HTTPS
  return 301 https://$host$request_uri;
}

server {
    # Listen HTTPS
    listen 443 ssl;
    server_name api.ezeugwagerrard.com www.api.ezeugwagerrard.com;

    # SSL config
    ssl_certificate /etc/letsencrypt/live/api.ezeugwagerrard.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/api.ezeugwagerrard.com/privkey.pem;

    # Proxy Config
    location / {
        proxy_pass http://44.208.88.155:1337;
        proxy_http_version 1.1;
        proxy_set_header X-Forwarded-Host $host;
        proxy_set_header X-Forwarded-Server $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_pass_request_headers on;

   }
}
